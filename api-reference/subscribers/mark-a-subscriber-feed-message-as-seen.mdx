```java Java
import co.novu.common.base.Novu;
import co.novu.api.subscribers.responses.SubscriberNotificationResponse;
import co.novu.api.subscribers.requests.MarkSubscriberFeedAsRequest;
import co.novu.api.subscribers.pojos.Mark;

public class Main {
    public static void main(String[] args) {
        String apiKey = "<NOVU_API_KEY>";
        Novu novu = new Novu(apiKey);
        String subscriberId = "<SUBSCRIBER_ID>";

        Mark mark = new Mark();
        mark.setRead(true);
        mark.setSeen(true);
        
        MarkSubscriberFeedAsRequest request = new MarkSubscriberFeedAsRequest();
        request.setMark(mark);
        request.setMessageId("<MESSAGE_ID>");

        SubscriberNotificationResponse response = novu.markSubscriberMessageFeedAs(request, subscriberId);
    }
}
```
