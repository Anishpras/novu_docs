Before you can start integrating Novu into your application, you need to create a Novu account and set up a new workflow in the Novu Dashboard. This guide walks you through those steps.

<Steps>
  <CreateAccountStep />

  <Step title="Install the Inbox React package">
    ```package-install 
    npm install @novu/notification-center
    ```
  </Step>

  <Step title="Create the Inbox component">
    To start using the Inbox component, you need to embed it in your application. 
    Import the <Method href="/docs/platform/inbox/react/components/inbox">{`<NovuProvider />`}</Method> and <Method href="/docs/platform/inbox/react/components/inbox">{`<PopoverNotificationCenter />`}</Method> components and pass the <Tooltip content="The application identifier is a unique identifier for your application. You can find it in the Novu Dashboard under the API keys page.">`applicationIdentifier`</Tooltip> and <Tooltip content="The subscriber ID is the unique identifier for the user in your application, typically the user's id in your database.">`subscriberId`</Tooltip> to the component.
      
    ```tsx title="app/components/inbox.tsx"
    import { 
      NovuProvider, 
      PopoverNotificationCenter,
      NotificationBell
    } from '@novu/notification-center';
    import { useNavigate } from '@remix-run/react';
    
    export function NotificationInbox() {
      const navigate = useNavigate();
      
      return (
        <NovuProvider
          applicationIdentifier="YOUR_APPLICATION_IDENTIFIER"
          subscriberId="YOUR_SUBSCRIBER_ID"
        >
          <PopoverNotificationCenter 
            onNotificationClick={(notification) => {
              if (notification.cta?.data?.url) {
                navigate(notification.cta.data.url);
              }
            }}
          >
            {({ unseenCount }) => <NotificationBell unseenCount={unseenCount} />}
          </PopoverNotificationCenter>
        </NovuProvider>
      );
    }
    ```
    <SignedOut>
      <div className="text-sm text-gray-500 text-center mt-[-12px]">
        [Sign in](https://dashboard-v2.novu.co/auth/sign-up) to get your own API keys
      </div>
    </SignedOut>
  
    <Callout type="info">
      You can find the `applicationIdentifier` in the Novu Dashboard under the [API keys page](https://dashboard-v2.novu.co/api-keys).
      The `subscriberId` is the unique identifier of the user in your application, learn more about subscribers [here](/docs/concepts/subscribers).
    </Callout>
  </Step>

  <Step title="Add the Inbox to your layout">
  Now you can import the `NotificationInbox` component and add it to your layout, it will automatically display the Bell icon with a notification counter.
  
    ```tsx title="app/root.tsx"
    import { NotificationInbox } from "~/components/inbox";

    export default function App() {
      return (
        <html>
          <body>
            <nav>
              <NotificationInbox />
            </nav>
            <Outlet />
          </body>
        </html>
      );
    }
    ```
  </Step>
