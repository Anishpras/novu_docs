Before you can start integrating Novu into your application, you need to create a Novu account and set up a new workflow in the Novu Dashboard. This guide walks you through those steps.

<Steps>
  <Step title="Install the Inbox package">
    ```package-install 
    npm install @novu/react 
    ```
  </Step>

  <Step title="Create the Inbox component">
    To start using the Inbox component, you need to embed it in your application. 
    Import the `Inbox` component and pass the `applicationIdentifier` and `subscriberId` to the component.

      
    ```tsx title="components/inbox.tsx"
    'use client';
    import React from 'react';
    import { Inbox } from '@novu/nextjs';
    import { useRouter } from 'next/navigation';
    
    export function NotificationInbox() {
      const router = useRouter();
      
      return (
        <Inbox // [!code highlight]
          applicationIdentifier="YOUR_APPLICATION_IDENTIFIER" // [!code highlight]
          subscriberId="YOUR_SUBSCRIBER_ID" // [!code highlight]
          routerPush={(path: string) => router.push(path)} // [!code highlight]
        /> // [!code highlight]
      );
    }
    ```
    <SignedOut>
      <div className="text-sm text-gray-500 text-center mt-[-12px]">
        [Sign in](https://dashboard-v2.novu.co/auth/sign-up) to get your own API keys
      </div>
    </SignedOut>
  
    <Callout type="info">
      You can find the `applicationIdentifier` in the Novu Dashboard under the [API keys page](https://dashboard-v2.novu.co/api-keys).
      The `subscriberId` is the unique identifier of the user in your application, learn more about subscribers [here](/docs/concepts/subscribers).
    </Callout>

    Novu uses the `routerPush` prop to make your notifications navigatable. We will automatically pass the `redirect.url` from your workflow definitions to the `routerPush` prop.

    <Accordions>
      <Accordion title="European Region Instructions">
        If you are using the European region in novu, you need to set the `backendUrl` and `socketUrl` props.
        ```tsx
        import { Inbox } from '@novu/react';

        function Novu() {
          return (
            <div>
              <Inbox
                applicationIdentifier="YOUR_APPLICATION_IDENTIFIER"
                subscriberId="YOUR_SUBSCRIBER_ID"
                backendUrl="https://eu.api.novu.co"
                socketUrl="https://eu.ws.novu.co"
              />
            </div>
          );
        }
        ```
      </Accordion>
    </Accordions>
  </Step>

  <Step title="Add the Inbox to your navbar">
  Now you can import the `NotificationInbox` component and add it to your navbar, it will automatically display the Bell icon, you can customize it later.
  
    ```tsx title="pages/layout.tsx"
    import { useRouter } from 'next/navigation';
    import { NotificationInbox } from "./components/inbox";

    export default function Layout() {
      return (
        <nav>
       // [!code highlight]
          <NotificationInbox /> 
        </nav>
      );
    }
    ```
  </Step>

    </Button>
  </Step>
</Steps>
